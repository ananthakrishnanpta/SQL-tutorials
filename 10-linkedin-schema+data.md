DROP DATABASE IF EXISTS linkedin_realistic;
CREATE DATABASE linkedin_realistic;
USE linkedin_realistic;

CREATE TABLE company(id INT PRIMARY KEY AUTO_INCREMENT,name VARCHAR(200),alias VARCHAR(100),website VARCHAR(100),headquarters VARCHAR(100),established YEAR);
CREATE TABLE locations(id INT PRIMARY KEY AUTO_INCREMENT,name VARCHAR(100),country VARCHAR(100));
CREATE TABLE company_branch(id INT PRIMARY KEY AUTO_INCREMENT,company_id INT,location_id INT,FOREIGN KEY(company_id) REFERENCES company(id),FOREIGN KEY(location_id) REFERENCES locations(id));
CREATE TABLE account(id INT PRIMARY KEY AUTO_INCREMENT,role ENUM('regular','recruiter'),first_name VARCHAR(100),middle_name VARCHAR(100),last_name VARCHAR(100),password VARCHAR(100),dob DATE,bio TEXT,email VARCHAR(100));
CREATE TABLE skill(id INT PRIMARY KEY AUTO_INCREMENT,title VARCHAR(100));
CREATE TABLE account_skill(id INT PRIMARY KEY AUTO_INCREMENT,account_id INT,skill_id INT,expertise ENUM('beginner','fluent','expert'),FOREIGN KEY(account_id) REFERENCES account(id),FOREIGN KEY(skill_id) REFERENCES skill(id));
CREATE TABLE job_posting(id INT PRIMARY KEY AUTO_INCREMENT,title VARCHAR(100),company_branch_id INT,vacancies INT,jd TEXT,added_at DATETIME DEFAULT CURRENT_TIMESTAMP,FOREIGN KEY(company_branch_id) REFERENCES company_branch(id));
CREATE TABLE job_posting_skill(id INT PRIMARY KEY AUTO_INCREMENT,job_posting_id INT,skill_id INT,FOREIGN KEY(job_posting_id) REFERENCES job_posting(id),FOREIGN KEY(skill_id) REFERENCES skill(id));
CREATE TABLE application(id INT PRIMARY KEY AUTO_INCREMENT,applicant_id INT,job_posting_id INT,applied_at DATETIME DEFAULT CURRENT_TIMESTAMP,status ENUM('applied','shortlisted','selected','rejected'),FOREIGN KEY(applicant_id) REFERENCES account(id),FOREIGN KEY(job_posting_id) REFERENCES job_posting(id));
